function calculatePetalSize(
  productCount: number,
  allCounts: number[],
  config: PetalConfig
): number {
  const { minSize, maxSize, sizeMode } = config
  const maxCount = Math.max(...allCounts)

  let normalized: number

  switch (sizeMode) {
    case 'linear':
      normalized = productCount / maxCount
      break
    case 'sqrt':
      normalized = Math.sqrt(productCount) / Math.sqrt(maxCount)
      break
    case 'log':

      normalized = productCount / maxCount

}
